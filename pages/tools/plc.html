    // Simple Start/Stop Circuit (Ladder Logic)
    ( )  --[ ]--( )  Start Button
    ( )  --[\]--( )  Stop Button
    ( )  --[ ]--( )  Motor Output


    // Equivalent Structured Text (ST)
    IF Start_Button THEN
        Motor := TRUE;
    END_IF;

    IF Stop_Button THEN
        Motor := FALSE;
    END_IF;


    // Function Block Diagram (FBD)
    (Start_Button) ---> [ OR ] ---> (Motor)
    (Motor)        ---> [ AND ] ---> (Motor)
    (Stop_Button)  ---> [ NOT ] ---> (AND)
